scr_bar_support

Description

Adds a supplemental bar to the figure, if relevant to the data

Usage

scr_bar_support(
  core_frame,
  color = rgb(0.8, 0.8, 0.8, alpha = 0.25),
  alpha = 1,
  guide_line = NULL,
  guide_line_type = 1,
  guide_line_size = 1,
  guide_line_color = "black",
  mapping = NULL,
  label = "",
  styler = NA,
  width = 0.8,
  adj = 0.5
)

core_frame

color

from base
