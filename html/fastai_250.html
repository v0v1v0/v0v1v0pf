  dl = NULL,
  with_input = FALSE,
  with_decoded = FALSE,
  cbs = NULL,
  reorder = TRUE
)

Arguments

ds_idx

dataset index
dl

DL application
with_input

with input or not
with_decoded

with decoded or not
cbs

callbacks
reorder

reorder or not

Value

None
